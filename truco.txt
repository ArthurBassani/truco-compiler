<STARTPROGRAMA>	::=			"tresehojogo"

<ABRECHA>		::= 		"desce"
<FECHAVE>		::= 		"sobe"

<ABREPAR>		::= 		"passa"
<FECHAPAR>  	::= 		"sinal"

<DECLASSE>		::=			"carteado"
<MODPRIVADO>	::=			"privado"
<MODPUBLICO>	::=			"publico"
<THIS>			::=			"eu"
<NEW> 			::=			"novinho"

<DECLFUNC> 		::= 		"vai"
<CHAMAFUNC>		::=			"vem"
<RETURN>		::=			"facao"

<ATRIVAR>		::=			"receba"
<FIMINST>		::=			"pato"

<TIPOINT>   	::= 		"copa"
<TIPODOUBLE>    ::= 		"gato"
<TIPOBOOLEAN>   ::= 		"mole"
<TIPOSTRING> 	::=			"espadilha"
<TIPOVETOR>		::=			"manilha"

<BOOLTRUE>		::=			"joga"
<BOOLFALSE>		::=			"corre"
<NULL>			::=			"null"
<ACESSVETOR>	::=			":"

<SINALDECIMAL>	::=			"."
<SINALSTRING>	::=			"'"

<SOMA>			::= 		"mais"
<SUBTRACAO>		::= 		"menos"
<DIVISAO> 		::= 		"div"
<MULTIPLICACAO>	::= 		"mul"

<OPAND>			::=			"e"
<OPOR>			::=			"ou"
<OPMAIOR>      	::= 		"maior"
<OPMENOR>      	::= 		"menor"
<OPIGUAL>      	::= 		"igual"
<OPDIF>        	::= 		"diferente"

<COND>			::=			"parceiro"
<FOR> 			::=			"torna"
<REP3>			::=			"come3"

<IDENT> 		::=			( <LETRA> )( <LETRA> | <DIGITO> )*
<LETRA>			::=			[ "A"-"Z" ] | [ "a"-"z" ]
<STRING> 		::= 		"'" (~["'", "\n", "\r"])+ "'"

<DECIMAL>		::=			( <INTEIRO> <SINALDECIMAL> <INTEIRO> )
<INTEIRO> 		::= 		( <DIGITO> )+
<#DIGITO> 		::= 		[ "0"-"9" ]

# QUALQUER OUTRA ENTRADA CAI AQUI
<INVALIDO> 		::= 		~[]

<MAIN> 			::= 		<STARTPROGRAMA> <ABRECHA>
								(<BLOCO>)*
							<FECHAVE>

<OPARIT>		::= 		( <SOMA>
							| <SUBTRACAO>
							| <MULTIPLICACAO>
							| <DIVISAO>
							)

<OPLOG>			::= 		( <OPMAIOR>
							| <OPMENOR>
							| <OPIGUAL>
							| <OPDIF>
							)

<OPCOND>		::=			( <OPAND> | <OPOR> )

<EXPARIT>		::=			<TERMO> ( <OPARIT> <TERMO> )*

<EXPLOG>		::= 		<EXPARIT> <OPLOG> <EXPARIT>

<EXPCOND>		::= 		<EXPLOG> ( <OPCOND>  ( <ABREPAR> <EXPCOND> <FECHAPAR> | <EXPLOG>) )*

<TIPODADO>		::=			( <TIPOINT>
							| <TIPODOUBLE>
							| <TIPOBOOLEAN>
							| <TIPOSTRING>
							| tipoVetor()
							)

<TIPOVETOR>		::=			<TIPOVETOR> tipoDado()

<IDENTIFICADOR> ::= 		<IDENT> [ <ACESSAVETOR> ] ( <ATRIBUI> <FIMINST> )

<ACESSAVETOR>	::=			<ACESSVETOR> ( <IDENT> | <INTEIRO> )

<DECLVAR>		::=			<TIPODADO>	<IDENT> [ <ATRIBUI> ] <FIMINST>

<NUMERO>		::=			[ <SOMA> | <SUBTRACAO> ] ( <DECIMAL> | <INTEIRO> )

<TERMO>			::=			( <IDENT> [ <ACESSAVETOR> ]
							| <NULL>
							| <BOOLTRUE>
							| <BOOLFALSE>
							| <STRING>
							| <NUMERO>
							| <_CHAMAFUNCAO>
							)


<ATRIBUI>		::= 		<ATRIVAR> (  <ATRIBUIVETOR> | <EXPARIT> )

<ATRIBUIVETOR>  ::=			<ABRECHA> <TERMO> ( <FIMINST> <TERMO> )* <FECHAVE>

<CONDICIONAL>  	::= 		<COND> <ABREPAR> <EXPCOND> <FECHAPAR>
								<ABRECHA>
									(<BLOCO>)*
								<FECHAVE>

<LACOREP>		::= 		<FOR> <ABREPAR> <BLOCO> <EXPLOG> <FIMINST> <BLOCO> <FECHAPAR>
								<ABRECHA>
									(<BLOCO>)*
								<FECHAVE>

<LACOREP3>		::= 		<REP3> <ABRECHA>
								(<BLOCO>)*
							<FECHAVE>

<RETORNA>		::=			<RETURN> <EXPARIT> <FIMINST>

<_CHAMAFUNCAO> 	::=			<CHAMAFUNC> <IDENT> <ABREPAR> termo() ( <FIMINST> termo() )* <FECHAPAR>

<CHAMAFUNCAO>	::= 		<_CHAMAFUNCAO> <FIMINST>

<DECLFUNCAO>	::= 		<DECLFUCN> <TIPODADO> <IDENT>
								<ABREPAR> (<DECLVAR>)* <FECHAPAR>
								<ABRECHA>
									(<BLOCO>)*
								<FECHAVE>

<BLOCO>			::=			( <DECLVAR>
							| <IDENTIFICADOR
							| <CONDICIONAL>
							| <LACOREP>
							| <LACOREP3
							| <DECLFUNCAO>
							| <RETORNA>
							| <CHAMAFUNCAO>
							)
